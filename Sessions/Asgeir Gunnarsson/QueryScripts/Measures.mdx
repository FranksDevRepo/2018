SELECT [CATALOG_NAME] ,
      	[CUBE_NAME] ,
       [MEASUREGROUP_NAME] ,
       [MEASURE_DISPLAY_FOLDER] ,
       [MEASURE_CAPTION] ,
       [MEASURE_IS_VISIBLE] ,
       [EXPRESSION],
       [DEFAULT_FORMAT_STRING] ,
       [MEASURE_AGGREGATOR]
FROM   $SYSTEM.MDSCHEMA_MEASURES
WHERE   CUBE_NAME = 'Financial Performance'
ORDER BY [MEASURE_CAPTION]
